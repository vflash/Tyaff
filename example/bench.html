🛠 Сравнение memo API

Как предотвратить лишний рендер дочернего компонента, когда поменялись только props родителя

tyaff

const Item = Component({
  memo(p) {
    return [p.value];
  },
  render(p) {
    return h('div', null, p.value);
  }
});

React

const Item = React.memo(
  function Item({ value }) {
    return rh('div', null, value);
  }
